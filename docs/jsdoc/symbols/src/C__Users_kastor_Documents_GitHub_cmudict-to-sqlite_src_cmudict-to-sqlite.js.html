 10 /**
 11  * @fileOverview A utility for parsing the CMU Pronouncing Dictionary into an sqlite database using Node.js

 20 /**
 21  * Reads the cmudict into a JavaScript array
 22  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 23  * @version 20130517
 24  * @requires fs
 25  * @function
 26  * @param {String} fileName The name and location of the cmudict to parse.
 27  * @example
 28  * // Reading the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 29  * // a JavaScript array:
 30  * 
 31  * var cmu = require('cmudict-to-sqlite');
 32  * var fileName = 'cmudict.0.7a';
 33  * var cmudict = cmu.cmudictToArray(fileName);
 34  * // the array will be huge and on my machine it takes longer to display it to 
 35  * // the console than it does to parse the file so, let's just display the first 
 36  * // thirty records
 37  * var count = 30;
 38  * while (count > 0) {
 39  *     console.log(cmudict[count]);
 40  *     count -= 1;
 41  * }
 42  */
 43 function cmudictToArray (fileName) {
 44     'use strict';
 45     var fs = require('fs');
 46     var cmudict = fs.readFileSync(fileName, 'utf8');
 47     
 48     cmudict = cmudict.replace(/;;;.*/g, ''); // strip comments
 49     cmudict = cmudict.replace(/(\r\n|\r|\n)/g, '\n'); // normalize line endings
 50     cmudict = cmudict.trim(); // remove superfluous white space
 51     cmudict = cmudict.split('\n'); // split into array of lines
 52     cmudict = cmudict.map(function(line) {
 53         return line.split('  ');
 54     }); // array items subdivided into word @ 0 code @ 1
 55     
 56     return cmudict;
 57 }

 59 /**
 60  * Parses the cmudict into an SQLite database
 61  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 62  * @version 20130517
 63  * @requires sqlite3
 64  * @requires cmudictToArray
 65  * @function
 66  * @param {String} fileName The name and location of the cmudict to parse.
 67  * @example
 68  * // Converting the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 69  * // an sqlite database:
 70  * 
 71  * var cmu = require('cmudict-to-sqlite');
 72  * var fileName = 'cmudict.0.7a';
 73  * cmu.cmudictToSqliteDb(fileName);
 74  */
 75 function cmudictToSqliteDb (fileName) {
 76     'use strict';
 77     var sqlite3 = require('sqlite3').verbose();
 78 
 79     var db = new sqlite3.Database(fileName + '.sqlite');
 80     db.exec('PRAGMA journal_mode = MEMORY;');
 81     
 82     db.exec("CREATE TABLE cmudict (word TEXT, code TEXT)", function () {
 83         var stmt = db.prepare("INSERT INTO cmudict VALUES (?, ?)");
 84         cmudictToArray(fileName).forEach(function(entry) {
 85             stmt.run(entry[0], entry[1]);
 86         });
 87         stmt.finalize();
 88         db.close();
 89     });
 90 }
 91 
 92 module.exports.cmudictToArray = cmudictToArray;
 93 module.exports.cmudictToSqliteDb = cmudictToSqliteDb;
