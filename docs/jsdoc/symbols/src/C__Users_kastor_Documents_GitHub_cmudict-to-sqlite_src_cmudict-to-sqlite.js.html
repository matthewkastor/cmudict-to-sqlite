 21 /**
 22  * Reads the cmudict file into a JavaScript array
 23  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 24  * @version 20130517
 25  * @requires fs
 26  * @function
 27  * @param {String} fileName The name and location of the cmudict to parse.
 28  * @example
 29  * // Reading the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 30  * // a JavaScript array:
 31  * 
 32  * var cmu = require('cmudict-to-sqlite');
 33  * var fileName = 'cmudict.0.7a';
 34  * var cmudict = cmu.cmudictToArray(fileName);
 35  * // the array will be huge and on my machine it takes longer to display it to 
 36  * // the console than it does to parse the file so, let's just display the first 
 37  * // thirty records
 38  * var count = 30;
 39  * while (count > 0) {
 40  *     console.log(cmudict[count]);
 41  *     count -= 1;
 42  * }
 43  */
 44 function cmudictToArray (fileName) {
 45     'use strict';
 46     var fs = require('fs');
 47     var cmudict = fs.readFileSync(fileName, 'utf8');
 48     
 49     cmudict = cmudict.replace(/;;;.*/g, ''); // strip comments
 50     cmudict = cmudict.replace(/(\r\n|\r|\n)/g, '\n'); // normalize line endings
 51     cmudict = cmudict.trim(); // remove superfluous white space
 52     cmudict = cmudict.split('\n'); // split into array of lines
 53     cmudict = cmudict.map(function(line) {
 54         return line.split('  ');
 55     }); // array items subdivided into word @ 0 code @ 1
 56     
 57     return cmudict;
 58 }

 60 /**
 61  * Parses the cmudict file into an SQLite database.
 62  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 63  * @version 20130517
 64  * @requires sqlite3
 65  * @requires cmudictToArray
 66  * @function
 67  * @param {String} fileName The name and location of the cmudict to parse.
 68  * @example
 69  * // Converting the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 70  * // an sqlite database:
 71  * 
 72  * var cmu = require('cmudict-to-sqlite');
 73  * var fileName = 'cmudict.0.7a';
 74  * cmu.cmudictToSqliteDb(fileName);
 75  */
 76 function cmudictToSqliteDb (fileName) {
 77     'use strict';
 78     var sqlite3 = require('sqlite3').verbose();
 79     var sql = {
 80         "journalMode"   :   "PRAGMA journal_mode = MEMORY;",
 81         "insert"        :   "INSERT INTO cmudict VALUES (?, ?)",
 82         "dbCreate"      :   "CREATE TABLE cmudict ( " +
 83                                 "word TEXT PRIMARY KEY ASC " +
 84                                     "NOT NULL ON CONFLICT ABORT " +
 85                                     "UNIQUE ON CONFLICT ABORT, " +
 86                                 "code TEXT NOT NULL ON CONFLICT ABORT " +
 87                             ");"
 88     };
 89     var db = new sqlite3.Database(fileName + '.sqlite');
 90     db.exec(sql.journalMode);
 91     db.exec(sql.dbCreate, function () {
 92         var stmt = db.prepare(sql.insert);
 93         cmudictToArray(fileName).forEach(function(entry) {
 94             stmt.run(entry[0], entry[1]);
 95         });
 96         stmt.finalize();
 97         db.close();
 98     });
 99 }

101 /**
102  * Represents the cmudict database and provides convenience methods for common 
103  *  tasks. When creating a new instance of the 
104  *  CmudictDb the given database file will be opened and several prepared 
105  *  statements will be loaded. Be sure to call the unload method when you are 
106  *  finished using the database so that the database may be closed properly.
107  * @class Represents the cmudict database and provides convenience methods for 
108  * common tasks.
109  * @param {String} cmudictFile Optional. The name and location of the cmudict 
110  *  database generated with cmudictToSqliteDb. Defaults to cmudict.0.7a.sqlite.
111  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
112  * @version 20130518
113  * @see cmudictToSqliteDb
114  * @requires sqlite3
115  */
116 function CmudictDb (cmudictFile) {
117     'use strict';
118     var my = this;
119     var sqlite3 = require('sqlite3').verbose();
120     cmudictFile = cmudictFile || 'cmudict.0.7a.sqlite';
121     my.db = new sqlite3.Database(cmudictFile);
122     my.db.exec('PRAGMA journal_mode = MEMORY;');
123     my.preparedStatements = {
124         "lookupWord" : my.db.prepare("SELECT * FROM cmudict WHERE word IS ?"),
125         "lookupCode" : my.db.prepare("SELECT * FROM cmudict WHERE code IS ?"),
126         "fuzzyLookupWord" : my.db.prepare("SELECT * FROM cmudict WHERE word LIKE ?"),
127         "fuzzyLookupCode" : my.db.prepare("SELECT * FROM cmudict WHERE code LIKE ?"),
128         "addEntry" : my.db.prepare("INSERT INTO cmudict VALUES (?,?);"),
129         "updateWord" : my.db.prepare("UPDATE cmudict SET word=? WHERE word=?;"),
130         "updateCode" : my.db.prepare("UPDATE cmudict SET code=? WHERE code=?;"),
131         "deleteEntry" : my.db.prepare("DELETE FROM cmudict WHERE word=?;")
132     };
133 }
134 /**
135  * Finalizes all preparedStatements and closes the database.
136  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
137  * @version 20130518
138  */
139 CmudictDb.prototype.unload = function unload () {
140     'use strict';
141     var my = this;
142     Object.keys(my.preparedStatements).forEach(function (stmt) {
143         my.preparedStatements[stmt].finalize();
144     });
145     my.db.close();
146 };
147 /**
148  * Searches for the given word.
149  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
150  * @version 20130518
151  * @param {String} word The word to look for in the words field.
152  * @param {Function} callback The callback to execute when results have been 
153  *  retreived. Takes two arguments: error and rows, in that order.
154  * @example
155  * var cmu = require('cmudict-to-sqlite');
156  * cmu = new cmu.CmudictDb();
157  * cmu.lookupWord('zebra', function (err, rows) {
158  *     console.log('lookupWord Results');
159  *     if (err) { console.log(err); }
160  *     if (rows) { console.log(rows); }
161  * });
162  * // do other stuff with the database . . .
163  * cmu.unload();
164  */
165 CmudictDb.prototype.lookupWord = function lookupWord (word, callback) {
166     'use strict';
167     var my = this;
168     word = word.toUpperCase();
169     my.preparedStatements.lookupWord.all(word, callback);
170 };
