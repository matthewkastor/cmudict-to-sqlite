 21 /**
 22  * Reads the cmudict file into a JavaScript array of arrays.
 23  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 24  * @version 20130517
 25  * @requires fs
 26  * @function
 27  * @param {String} fileName The name and location of the cmudict to parse.
 28  * @returns {Array} An array of arrays. Each element in the
 29  *  array is an array with item 0 being the word and item 1 being the ARPAbet 
 30  *  code. The array will also have a license property which contains the license
 31  *  header from the given cmudict file.
 32  * @example
 33  * // Reading the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 34  * // a JavaScript array:
 35  * 
 36  * var cmu = require('cmudict-to-sqlite');
 37  * var fileName = 'cmudict.0.7a';
 38  * var cmudict = cmu.cmudictToArray(fileName);
 39  * // the array will be huge and on my machine it takes longer to display it to 
 40  * // the console than it does to parse the file so, let's just display the first 
 41  * // thirty records
 42  * var count = 30;
 43  * while (count > 0) {
 44  *     console.log(cmudict[count]);
 45  *     count -= 1;
 46  * }
 47  */
 48 function cmudictToArray (fileName) {
 49     'use strict';
 50     var fs = require('fs');
 51     var cmudict = fs.readFileSync(fileName, 'utf8');
 52     cmudict = cmudict.replace(/(\r\n|\r|\n)/g, '\n'); // normalize line endings
 53     var license = cmudict.match(/;;;.*/g).reduce(function (prev, curr){
 54         return prev + curr + '\n';
 55     },'');
 56     cmudict = cmudict.replace(/;;;.*/g, ''); // strip comments
 57     cmudict = cmudict.trim(); // remove superfluous white space
 58     cmudict = cmudict.split('\n'); // split into array of lines
 59     cmudict = cmudict.map(function(line) {
 60         return line.split('  ');
 61     }); // array items subdivided into word @ 0 code @ 1
 62     
 63     cmudict.license = license;
 64     return cmudict;
 65 }

 67 /**
 68  * Parses the cmudict file into an SQLite database.
 69  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 70  * @version 20130517
 71  * @requires sqlite3
 72  * @requires cmudictToArray
 73  * @function
 74  * @param {String} fileName The name and location of the cmudict to parse.
 75  * @see <a href="http://www.speech.cs.cmu.edu/cgi-bin/cmudict">
 76  *  Carnegie Mellon University's information and download links for cmudict</a>
 77  * @see <a href="http://en.wikipedia.org/wiki/CMU_Pronouncing_Dictionary">
 78  *  Wikipedia's article on the CMU Pronouncing Dictionary</a>
 79  * @see <a href="http://en.wikipedia.org/wiki/Arpabet">
 80  *  Wikipedia's ARPAbet article</a>
 81  * @example
 82  * // Converting the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 83  * // an sqlite database:
 84  * 
 85  * var cmu = require('cmudict-to-sqlite');
 86  * var fileName = 'cmudict.0.7a';
 87  * cmu.cmudictToSqliteDb(fileName);
 88  */
 89 function cmudictToSqliteDb (fileName) {
 90     'use strict';
 91     var cmudict = cmudictToArray(fileName);
 92     var sqlite3 = require('sqlite3').verbose();
 93     var sql = {
 94         "journalMode"   :   "PRAGMA journal_mode = MEMORY;",
 95         "insert"        :   "INSERT INTO cmudict VALUES (?, ?)",
 96         "insertMetadata":   "INSERT INTO metadata VALUES (?, ?)",
 97         "cmuDictTable"  :   "CREATE TABLE cmudict ( " +
 98                                 "word TEXT PRIMARY KEY ASC " +
 99                                     "NOT NULL ON CONFLICT ABORT " +
100                                     "UNIQUE ON CONFLICT ABORT, " +
101                                 "code TEXT NOT NULL ON CONFLICT ABORT " +
102                             ");",
103         "metaTable"     :   "CREATE TABLE metadata ( " +
104                                 "name TEXT PRIMARY KEY ASC " +
105                                     "NOT NULL ON CONFLICT ABORT " +
106                                     "UNIQUE ON CONFLICT ABORT, " +
107                                 "data TEXT NOT NULL ON CONFLICT ABORT " +
108                             ");"
109     };
110     var metadata = {
111         "license" : cmudict.license
112     };
113     var db = new sqlite3.Database(fileName + '.sqlite');
114     db.exec(sql.journalMode, function (err) {                // set journal mode
115         if(err) { console.log(err); }
116         db.exec(sql.metaTable, function (err) {         // create metadata table
117             if(err) { console.log(err); }
118             var stmt = db.prepare(sql.insertMetadata);
119             Object.keys(metadata).forEach(function(name) {    // insert metadata
120                 stmt.run(name, metadata[name]);
121             });
122             stmt.finalize();
123             db.exec(sql.cmuDictTable, function (err) {   // create cmudict table
124                 if(err) { console.log(err); }
125                 var stmt = db.prepare(sql.insert);     // insert cmudict entries
126                 cmudict.forEach(function(entry) {
127                     stmt.run(entry[0], entry[1]);
128                 });
129                 stmt.finalize();
130                 db.close();                                    // close database
131             });
132         });
133     });
134 }
135 /**
136  * Represents the cmudict database and provides convenience methods for common 
137  *  tasks. When creating a new instance of the 
138  *  CmudictDb the given database file will be opened and several prepared 
139  *  statements will be loaded. Be sure to call the unload method when you are 
140  *  finished using the database so that the database may be closed properly. 
141  *  The dictionary is stored in the cmudict table and information, including 
142  *  the license for the CMU Pronunciation Dictionary itself, is stored in the 
143  *  metadata table.
144  * @class Represents the cmudict database and provides convenience methods for 
145  * common tasks.
146  * @param {String} cmudictFile Optional. The name and location of the cmudict 
147  *  database generated with cmudictToSqliteDb. Defaults to cmudict.0.7a.sqlite.
148  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
149  * @version 20130518
150  * @see cmudictToSqliteDb
151  * @see <a href="http://www.speech.cs.cmu.edu/cgi-bin/cmudict">
152  *  Carnegie Mellon University's information and download links for cmudict</a>
153  * @see <a href="http://en.wikipedia.org/wiki/CMU_Pronouncing_Dictionary">
154  *  Wikipedia's article on the CMU Pronouncing Dictionary</a>
155  * @see <a href="http://en.wikipedia.org/wiki/Arpabet">
156  *  Wikipedia's ARPAbet article</a>
157  * @requires sqlite3
158  */
159 function CmudictDb (cmudictFile) {
160     'use strict';
161     var my = this;
162     var sqlite3 = require('sqlite3').verbose();
163     cmudictFile = cmudictFile || 'cmudict.0.7a.sqlite';
164     my.db = new sqlite3.Database(cmudictFile);
165     my.db.exec('PRAGMA journal_mode = MEMORY;');
166     my.preparedStatements = {
167         "saveAsText" : my.db.prepare("SELECT * FROM cmudict;"),
168         "lookupMetadata" : my.db.prepare("SELECT * FROM metadata WHERE name=?;"),
169         "lookupWord" : my.db.prepare("SELECT * FROM cmudict WHERE word IS ?;"),
170         "lookupCode" : my.db.prepare("SELECT * FROM cmudict WHERE code IS ?;"),
171         "fuzzyLookupWord" : my.db.prepare("SELECT * FROM cmudict WHERE word LIKE ?;"),
172         "fuzzyLookupCode" : my.db.prepare("SELECT * FROM cmudict WHERE code LIKE ?;"),
173         "addEntry" : my.db.prepare("INSERT INTO cmudict VALUES (?,?);"),
174         "addMetadata" : my.db.prepare("INSERT INTO metadata VALUES (?,?);"),
175         "updateWord" : my.db.prepare("UPDATE cmudict SET word=? WHERE word=?;"),
176         "updateMetadata" : my.db.prepare("UPDATE metadata SET data=? WHERE name=?;"),
177         "updateCode" : my.db.prepare("UPDATE cmudict SET code=? WHERE code=?;"),
178         "deleteEntry" : my.db.prepare("DELETE FROM cmudict WHERE word=?;"),
179         "deleteMetadata" : my.db.prepare("DELETE FROM metadata WHERE name=?;")
180     };
181 }
182 /**
183  * Finalizes all preparedStatements and closes the database.
184  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
185  * @version 20130518
186  */
187 CmudictDb.prototype.unload = function unload () {
188     'use strict';
189     var my = this;
190     Object.keys(my.preparedStatements).forEach(function (stmt) {
191         my.preparedStatements[stmt].finalize();
192     });
193     my.db.close();
194 };
195 /**
196  * Saves the contents of the database as a text file using the same format as the 
197  *  original cmudict. The contents of metadata.license will be prepended to the 
198  *  beginning of the file, change it if you want to add information to the 
199  *  license.
200  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
201  * @version 20130519
202  * @param {String} fileName The name and location of the output file.
203  * @param {Function} callback The callback to execute after the file is 
204  *  written. The callback will be called with one argument, the string which 
205  *  was written to file.
206  * @example
207  *  var cmu = require('cmudict-to-sqlite');
208  *  cmu = new cmu.CmudictDb();
209  *  cmu.saveAsText('cmudict.txt', function(content) {
210  *      // if you want to see what was written to the file
211  *      // uncomment the next line.
212  *      // console.log(content);
213  *      // do other stuff with the database . . .
214  *      cmu.unload();
215  *  });
216  */
217 CmudictDb.prototype.saveAsText = function saveAsText (fileName, callback) {
218     'use strict';
219     var my = this;
220     var fs = require('fs');
221     function writeToFile (fileName, output, callback) {
222         fs.writeFile(fileName, output, {'encoding' : 'utf8'}, function (err) {
223             if(err) {
224                 throw err;
225             } else {
226                 if(callback) { callback(output); }
227             }
228         });
229     }
230     function compileDictionary (license) {
231         my.preparedStatements.saveAsText.all(function (err, rows) {
232             var output = license || '';
233             if (err) { console.log(err); }
234             if (rows) {
235                 output += '\n';
236                 output += rows.reduce(function (prev, curr) {
237                     return prev + curr.word + '  ' + curr.code + '\n';
238                 }, '');
239                 writeToFile(fileName, output, callback);
240             }
241         });
242     }
243     my.preparedStatements.lookupMetadata.all('license', function (err, rows) {
244         if (err) { console.log(err); }
245         if (rows) {
246             compileDictionary(rows[0].data.trim());
247         }
248     });
249 };
