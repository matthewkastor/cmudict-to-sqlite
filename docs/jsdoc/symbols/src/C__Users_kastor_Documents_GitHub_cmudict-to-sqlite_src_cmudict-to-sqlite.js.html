 21 /**
 22  * Reads the cmudict file into a JavaScript array
 23  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 24  * @version 20130517
 25  * @requires fs
 26  * @function
 27  * @param {String} fileName The name and location of the cmudict to parse.
 28  * @example
 29  * // Reading the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 30  * // a JavaScript array:
 31  * 
 32  * var cmu = require('cmudict-to-sqlite');
 33  * var fileName = 'cmudict.0.7a';
 34  * var cmudict = cmu.cmudictToArray(fileName);
 35  * // the array will be huge and on my machine it takes longer to display it to 
 36  * // the console than it does to parse the file so, let's just display the first 
 37  * // thirty records
 38  * var count = 30;
 39  * while (count > 0) {
 40  *     console.log(cmudict[count]);
 41  *     count -= 1;
 42  * }
 43  */
 44 function cmudictToArray (fileName) {
 45     'use strict';
 46     var fs = require('fs');
 47     var cmudict = fs.readFileSync(fileName, 'utf8');
 48     
 49     cmudict = cmudict.replace(/;;;.*/g, ''); // strip comments
 50     cmudict = cmudict.replace(/(\r\n|\r|\n)/g, '\n'); // normalize line endings
 51     cmudict = cmudict.trim(); // remove superfluous white space
 52     cmudict = cmudict.split('\n'); // split into array of lines
 53     cmudict = cmudict.map(function(line) {
 54         return line.split('  ');
 55     }); // array items subdivided into word @ 0 code @ 1
 56     
 57     return cmudict;
 58 }

 60 /**
 61  * Parses the cmudict file into an SQLite database.
 62  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
 63  * @version 20130517
 64  * @requires sqlite3
 65  * @requires cmudictToArray
 66  * @function
 67  * @param {String} fileName The name and location of the cmudict to parse.
 68  * @example
 69  * // Converting the CMU Pronouncing Dictionary from the flat file `cmudict` into 
 70  * // an sqlite database:
 71  * 
 72  * var cmu = require('cmudict-to-sqlite');
 73  * var fileName = 'cmudict.0.7a';
 74  * cmu.cmudictToSqliteDb(fileName);
 75  */
 76 function cmudictToSqliteDb (fileName) {
 77     'use strict';
 78     var sqlite3 = require('sqlite3').verbose();
 79 
 80     var db = new sqlite3.Database(fileName + '.sqlite');
 81     db.exec('PRAGMA journal_mode = MEMORY;');
 82     
 83     db.exec("CREATE TABLE cmudict (word TEXT, code TEXT)", function () {
 84         var stmt = db.prepare("INSERT INTO cmudict VALUES (?, ?)");
 85         cmudictToArray(fileName).forEach(function(entry) {
 86             stmt.run(entry[0], entry[1]);
 87         });
 88         stmt.finalize();
 89         db.close();
 90     });
 91 }

 93 /**
 94  * Represents the cmudict database and provides convenience methods for common 
 95  *  tasks. When creating a new instance of the 
 96  *  CmudictDb the given database file will be opened and several prepared 
 97  *  statements will be loaded. Be sure to call the unload method when you are 
 98  *  finished using the database so that the database may be closed properly.
 99  * @class Represents the cmudict database and provides convenience methods for 
100  * common tasks.
101  * @param {String} cmudictFile The name and location of the cmudict database 
102  *  generated with cmudictToSqliteDb.
103  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
104  * @version 20130518
105  * @see cmudictToSqliteDb
106  * @requires sqlite3
107  */
108 function CmudictDb (cmudictFile) {
109     var my = this;
110     var sqlite3 = require('sqlite3').verbose();
111     var cmudictFile = cmudictFile || 'cmudict.0.7a.sqlite';
112     my.db = new sqlite3.Database(cmudictFile);
113     my.db.exec('PRAGMA journal_mode = MEMORY;');
114     my.preparedStatements = {
115         "lookupWord" : my.db.prepare("SELECT * FROM cmudict WHERE word IS ?"),
116         "lookupCode" : my.db.prepare("SELECT * FROM cmudict WHERE code IS ?"),
117         "fuzzyLookupWord" : my.db.prepare("SELECT * FROM cmudict WHERE word LIKE ?"),
118         "fuzzyLookupCode" : my.db.prepare("SELECT * FROM cmudict WHERE code LIKE ?")
119     };
120 }
121 /**
122  * Finalizes all preparedStatements and closes the database.
123  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
124  * @version 20130518
125  */
126 CmudictDb.prototype.unload = function unload () {
127     var my = this;
128     Object.keys(my.preparedStatements).forEach(function (stmt) {
129         my.preparedStatements[stmt].finalize();
130     });
131     my.db.close();
132 };
133 /**
134  * Searches for the given word.
135  * @author <a href="mailto:matthewkastor@gmail.com">Matthew Kastor</a>
136  * @version 20130518
137  * @param {String} word The word to look for in the words field.
138  * @param {Function} callback The callback to execute when results have been 
139  *  retreived. Takes two arguments: error and rows, in that order.
140  * @example
141  * var cmu = require('cmudict-to-sqlite');
142  * cmu = new cmu.CmudictDb();
143  * cmu.lookupWord('zebra', function (err, rows) {
144  *     console.log('lookupWord Results');
145  *     console.log(rows);
146  * });
147  * // do other stuff with the database . . .
148  * cmu.unload();
149  */
150 CmudictDb.prototype.lookupWord = function lookupWord (word, callback) {
151     var my = this;
152     word = word.toUpperCase();
153     my.preparedStatements.lookupWord.all(word, callback);
154 };
